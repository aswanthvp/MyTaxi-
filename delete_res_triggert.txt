
CREATE TRIGGER [dbo].[delete_rest]
       ON [dbo].[trip_details]
AFTER delete
AS
 DECLARE @advance numeric(18,0),
		 @advance_us numeric(18,0),
		 @remarks numeric(18,0),
		 @diesel_cost numeric(18,0),
		@driver varchar(50),
		@company varchar(50),
		@km numeric(18,0),
		@cost numeric(5,0),
		@remarks_us numeric(18,0)
 
       SELECT @km=km,@cost=cost_km, @company=company,@driver=driver,@advance=advance, @advance_us=advance_us,@remarks=remarks,@diesel_cost=diesel_cost,@remarks_us=remarks_us
       FROM deleted

BEGIN
       SET NOCOUNT ON;
 
     update driver set pending_with_driver=pending_with_driver-@advance-@advance_us+@remarks+@remarks_us where driver=@driver
	 update company set total=total+@advance+@diesel_cost where company=@company
	 update company set total=total-(@km*@cost) where company=@company
	 update driver set payment=payment-(@km*@cost*.15) where driver=@driver
 
       
END
